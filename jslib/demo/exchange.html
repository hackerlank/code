jslib1.4.2兑换服务

目前互动活动的兑换有两种：用兑换码兑换奖品，用积分(某种资格)兑换奖品

用兑换码兑换奖品：

用积分(某种资格)兑换奖品：

function codeEx(){
    loader('service/exchange',function(){
        var excode = $app.G('excode').value;
		$app.exchange.code(excode);
	});
}

function scoreEx(id){
    loader('service/exchange',function(){
		$app.exchange.score(id);
	});
}
