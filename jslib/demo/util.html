jslib1.4.2工具类组件函数

1、获取URL的Get参数，如当前浏览器的url是"http://test.qq.com?id=1",这样我们一般可以如下代码获得ID的值：

function tUtilGet(){
	var id = $app.util.getURLParam('id');
	alert('当前url上的参数id的值是：'+id);
}

2、去除字符串首尾的空白字符，先输入前后有空白的文字，然后点击测试看看， 

function tUtilTrim(){
	var val = $app.G('tUtilInput').value;
	val = $app.util.trim(val);
	if(val == ''){
		alert('你没有输入！');
	}
	else{
		alert('你输入的去掉前后空白后的文字是：'+val);
	}
}

3、将文本拷贝到剪贴板，点击这里我们为你复制了一段话："hello world!"

function tUtilCopy(){
	var txt = "hello world!";
	$app.util.copyToClipboard(txt);
}

4、cookie操作，设置cookie 修改cookie 删除cookie

function tCookieSet(){
	$app.util.cookie("jslib", "1.4",{"domain":"appmedia.qq.com","path":"/","expires":3600}); 
	alert("你已经为你的浏览器中设置了jslib为1.4的cookie，请利用工具查看!");
}

function tCookieModify(){
	$app.util.cookie("jslib", "1.4.2",{"domain":"appmedia.qq.com","path":"/","expires":3600}); 
	alert("浏览器中jslib为1.4的cookie修改了值为1.4.2，请利用工具查看!");
}

function tCookieDel(){
	$app.util.cookie("jslib", null); 
	alert("浏览器中jslib的cookie已经删除，请利用工具确认!");
}
